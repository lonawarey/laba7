#include <stdio.h>
#include <math.h>

int main() {
    double P;          // Процент увеличения
    double daily = 10; // Пробег в первый день (км)
    double total = 10; // Суммарный пробег
    int K = 1;         // Количество дней (начали с 1 дня)
    
    // Ввод данных
    printf("Введите процент увеличения P (0 < P < 50): ");
    scanf("%lf", &P);
    
    // Проверка корректности ввода
    if (P <= 0 || P >= 50) {
        printf("Ошибка: P должно быть в диапазоне 0 < P < 50\n");
        return 1;
    }
    
    // Рассчитываем дни до превышения 200 км
    while (total <= 200) {
        // Увеличиваем дневной пробег на P процентов
        daily = daily * (1 + P / 100);
        
        // Добавляем к суммарному пробегу
        total += daily;
        
        // Увеличиваем счетчик дней
        K++;
        
        // Защита от бесконечного цикла (на всякий случай)
        if (K > 365) {
            printf("За год не удалось превысить 200 км\n");
            return 1;
        }
    }
    
    // Вывод результатов
    printf("\nРезультаты:\n");
    printf("Количество дней K = %d\n", K);
    printf("Суммарный пробег S = %.2f км\n", total);
    
    // Дополнительная информация
    printf("\nПробег по дням:\n");
    
    // Повторяем расчет для вывода подробной информации
    total = 10;
    daily = 10;
    for (int i = 1; i <= K; i++) {
        printf("День %d: %.2f км (сумма: %.2f км)\n", i, daily, total);
        
        // Для последнего дня не нужно увеличивать пробег
        if (i < K) {
            daily = daily * (1 + P / 100);
            total += daily;
        }
    }
    
    return 0;
}
